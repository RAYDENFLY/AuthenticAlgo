def rl_threshold_optimizer(trading_environment):
    """
    Use RL untuk optimize confidence thresholds secara dynamic
    """
    agent = PPOTrainer(
        policy=ThresholdOptimizationPolicy(),
        env=trading_environment,
        learning_rate=0.001
    )
    
    # Train agent untuk maximize risk-adjusted returns
    best_thresholds = agent.train(
        total_timesteps=10000,
        callback=ThresholdCallback()
    )
    
    return best_thresholds
