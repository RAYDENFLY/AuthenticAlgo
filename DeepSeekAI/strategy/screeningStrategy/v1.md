            'position_size': '1-2% portfolio',

        },
        'low_confidence': {
            'min_confidence': 0.65,
            'position_size': '0.5-1% portfolio',
            'stop_loss': '0.8%',
            'take_profit': '1.5-2%'
        }
    }
    
    # Adjust based on market volatility
    if market_conditions['volatility'] == 'high':
        # Reduce position sizes in high volatility
        for level in entry_rules:
            entry_rules[level]['position_size'] *= 0.7
            entry_rules[level]['stop_loss'] *= 1.3
    
    return entry_rules
